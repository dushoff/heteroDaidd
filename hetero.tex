 \section{Homogeneous disease models}
\begin{frame}


\frametitle{Expanding our models}\begin{itemize}[<+-| alert@+>]

\item \textbf{Homogeneous} models assume everyone has the same:\begin{itemize}[<+-| alert@+>]
\item disease characteristics (e.g. susceptibility, tendency to
		transmit)
\item mixing rate
\item probability of mixing with each person\end{itemize}

\item \textbf{Heterogeneous} models allow people to be different\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{The basic reproductive number}\begin{itemize}[<+-| alert@+>]

\item $\Ro$ is the number of people who would be infected by an
	infectious individual \emph{in a fully susceptible population.}

\item $\Ro = \beta/\gamma = \beta D = (cp) D$\begin{itemize}[<+-| alert@+>]
\item $c$: Contact Rate
\item $p$: Probability of transmission (infectivity)
\item $D$: Average duration of infection\end{itemize}

\item A disease can invade a population if and only if $\Ro>1$.\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium}

\hfill\includegraphics[height=0.9\textheight,angle=270]{SIR_model_family/comp.pdf}\hfill\mbox{}\begin{itemize}[<+-| alert@+>]

\item Equilibrium is worth knowing even if the disease doesn't reach equilibrium

\item System will move around the equilibrium\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium analysis }\begin{itemize}[<+-| alert@+>]

\item $\Reffec$ is the number of people who would be infected by an
	infectious individual \emph{in a general population.}

\item $\ds\Reffec =  \Ro  \frac SN = pcD \frac SN$

\item At equilibrium:
		$\ds \Reffec = \Ro  \frac SN = 1. $

\item Thus:
		$\ds \frac SN = 1/R_0.  $

\item Proportion `affected' is $V = 1-S/N = 1-1/R_0$.\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Proportion affected}\begin{itemize}[<+-| alert@+>]

\item Proportion `affected' is $V = 1-S/N = 1-1/R_0$.\begin{itemize}[<+-| alert@+>]
\item * \uncover<+->{\color{blue} The same formula as the critical vaccination proportion!}
\item * \uncover<+->{\color{blue} If this proportion is made unavailable, the disease cannot spread}
\item * \uncover<+->{\color{blue} At least, in the homogeneous case}\end{itemize}\end{itemize}
\uncover<+->\mbox{}\end{frame}

 \section{Effects of heterogeneity}
\begin{frame}


\frametitle{Heterogeneity among hosts}\begin{itemize}[<+-| alert@+>]

\item Differences among people are pervasive, large and often correlated

\item We often consider transmission probability as the product of two
	components:\begin{itemize}[<+-| alert@+>]
\item The "infector" has tendency to infect $\tau$
\item The "recipient" has susceptibility $\sigma$\end{itemize}

\item Then $\Ro = p c D = (\sigma \tau) c D$,

\item Why do we assume this is multiplicative?\begin{itemize}[<+-| alert@+>]
\item * \uncover<+->{\color{blue} Convenience, question this assumption}\end{itemize}\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium calculations}\begin{itemize}[<+-| alert@+>]

\item Assume $p = \sigma\tau$ has a susceptibility component and a
	transmission component:\begin{itemize}[<+-| alert@+>]
\item $\Ro = \sigma\tau c D$
\item $\Reffec = \sigma\tau c D S/N$
\item Equilibrium $S/N = 1/\Ro$
\item Proportion affected: $1 - 1/\Ro$\end{itemize}\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium calculations with heterogeneity}\begin{itemize}[<+-| alert@+>]

\item $\tau D$ applies to infectious individuals $\rightarrow \tau_I
	D_I$

\item $\sigma$ applies to susceptible individuals $\rightarrow
	\sigma_S$

\item $c$ is complicated $\rightarrow c_S c_I/\bar c$\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium calculations with heterogeneity}\begin{itemize}[<+-| alert@+>]

\item $\Ro = \sigma_S\tau_I c_x D_I$
	measured during \emph{invasion}

\item $\Reffec = \sigma_S\tau_I c_x D_I S/N$
	measured at \emph{equilibrium}

\item Equilibrium $S/N \neq 1/\Ro$\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{How does $\Rx$ change?}\begin{itemize}[<+-| alert@+>]

\item Imagine a disease spread by people who differ only in their
	effective mixing rates

\item If the disease has just started spreading in the population, how do
	$c_S$ and $c_I$ compare to $\bar c$?\begin{itemize}[<+-| alert@+>]
\item $c_S \approx \bar c$; $c_I > \bar c$.\end{itemize}

\item If the disease is very widespread in the population?\begin{itemize}[<+-| alert@+>]
\item $c_S < \bar c$; $c_I \to \bar c$.\end{itemize}\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{The basic reproductive number}\begin{itemize}[<+-| alert@+>]

\item When the disease invades:\begin{itemize}[<+-| alert@+>]
\item The susceptible population $\approx$ the general population
\item The infectious population is likely to have higher values of $c$,
		$D$ and/or $\tau$\end{itemize}

\item $\Ro$ is typically greater than you would expect from a homogeneous
	model\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Equilibrium analysis}\begin{itemize}[<+-| alert@+>]

\item As disease prevalence goes up:\begin{itemize}[<+-| alert@+>]
\item Susceptible pool is the most resistant, or least exposed group
\item Infectious pool looks more like the general population.\end{itemize}

\item $\rightarrow$ lower proportion affected \emph{for a given value of
	\Ro.}\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{How diseases reach equilibrium }\begin{itemize}[<+-| alert@+>]

\item Diseases that invade have high values of \Ro

\item \Reffec\ must be 1 at equilibrium\begin{itemize}[<+-| alert@+>]
\item Potentially infectious contacts are wasted\begin{itemize}[<+-| alert@+>]
\item Many potential contacts are not susceptible (affected by
			disease)
\item Those not affected less susceptible than average\end{itemize}
\item Infectious pool less infectious\end{itemize}\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Spatial and network models}\begin{itemize}[<+-| alert@+>]

\item Individual-level, or spatial, heterogeneity also usually
	increases wasted contacts

\item Infectious people meet:\begin{itemize}[<+-| alert@+>]
\item people with similar social backgrounds
\item people with similar behaviours
\item people who are nearby geographically or in the contact network\end{itemize}

\item More wasted contacts further flatten the endemic curve\end{itemize}
\uncover<+->\mbox{}\end{frame}

\item $(1-P)^s$
\item $(1-P/s)^{\alpha s}$\end{itemize}\end{itemize}

\begin{frame}


\frametitle{Multi-group models}\begin{itemize}[<+-| alert@+>]

\item Divide the population into groups.\begin{itemize}[<+-| alert@+>]
\item cities and villages
\item rich and poor
\item high and low sexual activity
\item age, gender
\item ...\end{itemize}

\item Even if details are not correct, heterogeneity will emerge and move model in
	the right direction\end{itemize}
\uncover<+->\mbox{}\end{frame}

\begin{frame}


\frametitle{Individual-based models}

\hfill\includegraphics[width=0.8\textwidth]{my_images/network.png}\hfill\mbox{}\begin{itemize}[<+-| alert@+>]

\item Allow many possibilities:\begin{itemize}[<+-| alert@+>]
\item vary individual characteristics
\item add a network of interactions
\item let the network change\end{itemize}

\item Individual-based approaches require stochastic models\end{itemize}
\uncover<+->\mbox{}\end{frame}
